#!/usr/bin/perl
print <<help;
 This program converts a number from decimal to binary;
 It successively determines if the decimal value exceeds
 the pre-defined place values for the binary number system.
 
 If the tested value is in excess then a "1" is output,
 otherwise a "0" is output. In the event of a "1" the tested
 value is decremented (reduced) by the binary place value 
 and the process continues.

 The program deals only with values in the range 0-255 (ie
 an 8-bit binary number).
help

do {
 print "This program converts an integer (from the decimal\n";
 print " counting system) into a binary number.\n\n";
 print "Please enter a number in the range 0-255: ";
 my $number=<STDIN>;
 chomp $number; 
 print "\n\nThat number in binary is:\n";
 if ( ($number >=0) && ($number<=255) ) {
  if ($number>=128)  {
   print "1";
   $number=$number-128;
  }
  else  {
   print "0";
  }
  if ($number>=64) {
   print "1";
   $number=$number-64;
  }
  else  {
   print "0";
  }
  if ($number>=32)  {
   print "1";
   $number=$number-32;
  }
  else  {
   print "0";
  }  
  if ($number>=16)  {
   print "1";
   $number=$number-16;
  }
  else  {
   print "0";
  } 
  if ($number>=8)  {
   print "1";
   $number=$number-8;
  }
  else {
   print "0";
  }  
  if ($number>=4)  {
   print "1";
   $number=$number-4;
  }
  else  {
   print "0";
  }  
  if ($number>=2)  {
   print "1";
   $number=$number-2;
  }
  else  {
   print "0";
  }
  if ($number>=1)  {
   print "1";
   $number=$number-1;
  }
  else  {
   print "0";
  }
  print "\n";

 } #if
 else  {
  print "Invalid number entered.\n";
 }
}
