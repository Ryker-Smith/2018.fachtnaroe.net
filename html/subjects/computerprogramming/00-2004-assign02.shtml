// A program to dispaly the names, marks and grades of ten students
#include 
main ()
{
  int looper, loop2, loop3, mark;
  char grade, name[10];
  loop2 = 1;
  loop3 = 1;
  while (loop2 <=10)
  {
    printf ("Enter student name:");
    scanf ("%s", &name);
    printf ("Enter student mark:");
    scanf ("%d", &mark);
    if (mark <=49)
    grade = 'F';
    else
    if (mark <=64)
    grade = 'P';
    else
    if (mark <=79)
    grade = 'M';
    else
    if (mark <=100)
    grade = 'D';
    {
      if (mark >100)
      grade = '_', printf ("Error: please enter mark.\n");
      else
      if (mark <0)
      grade = '_', printf ("Error: please enter mark.\n");
      looper = looper + 1;
      loop3 = loop3 + 1;
    }
    printf ("Name\t");
    printf ("Mark\t");
    printf ("Grade\n");
    printf ("%s\t", name);
    printf ("%d\t", mark);
    printf ("%c\n", grade);
    looper = looper + 1;
    loop2 = loop2 + 1;
  }
}
