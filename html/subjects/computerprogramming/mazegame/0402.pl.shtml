#!/usr/bin/perl
use strict;
require Term::Screen;

my ($rt, @map, $left, $vert, $l);
# Prepare a new terminal screen
$rt=new Term::Screen;
$rt->clrscr();

# Get the contents of the map file
open (MF, "< maze04.map");
@map = <MF>;
close (MF);

$left=25;
$vert=5;
foreach $l (@map) {
  $rt->at($vert, $left)->puts("$l");
  $vert = $vert + 1;
}
