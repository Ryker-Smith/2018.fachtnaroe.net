Hidden input fields demo 01

This demo shows how input fields of type hidden may be used.

This is the input form:

<!DOCTYPE html>
<html>
<head>
<title>Hidden input demo #01</title>
</head>
<body>
<h2>Hidden input fields</h2>
<p>Press ^U to view source and see the values being sent</p>
<form action="cgi-bin/hidden_input_01.pl" method="post">
<input type="hidden" value="5" name="hfield1">
<input type="hidden" value="3" name="hfield2">
<input type="submit" value="Press to send the values">
</form>
</body>
</html>

This is the perl program:

#!/usr/bin/perl -T
use strict;
use CGI::Carp qw(fatalsToBrowser);
use CGI;

our (%in); # This is the variable used by CGI::ReadParse below
CGI::ReadParse(); # Tell the module to 'parse' the form 
my ($hfield1, $hfield2);
$hfield1 = $in{hfield1};
$hfield2 = $in{hfield2};
&content_type;
&html_top ("Hidden input demo result");
print "First field was $hfield1<br>\n";
print "Second field was $hfield2<br>\n";
print "The product is ", $hfield1*$hfield2,"<br>\n";
&html_tail;

sub content_type {
# standard html content-type string
  print "Content-Type: text/html\n\n";
}

sub html_top {
# print generic html header 
# Expects: String to use as a title value
  print "<!DOCTYPE html>\n<html><head>\n<title>";
  print @_;
  print "</title>\n";
  print "</head>\n<body>\n";
}

sub html_tail {
# print generic html ending
  print "\n</body>\n</html>\n";
}
