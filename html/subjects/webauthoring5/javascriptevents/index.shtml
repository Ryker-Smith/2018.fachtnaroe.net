Javascript Events

Copy and paste this into your own web location to see what it does - or click here.

<!DOCTYPE html>
<html>
<head>
<title>Simple javascript demo</title>
<meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />
<script type="text/javascript">
function mouse_over (nameofsomething) {
/* The element name is passed in to nameofsomething 
   I'm using a span in which the HTML can be re-written
*/
  document.getElementById(nameofsomething).innerHTML="Mouse is over";
}
function apples (dog) {
  document.getElementById(dog).innerHTML="You clicked";
}
function mouse_out (nameofsomething) {
  document.getElementById(nameofsomething).innerHTML="Mouse is out";
}
</script>
</head>
<body>
<div >
<span id="demo" onclick="apples('my_thing')" onmouseover="mouse_over('my_thing')" onmouseout="mouse_out('my_thing')">
Move your mouse <b>here</b> and ...
</span>
</div> 
<div id="my_thing">...this text will change in response to the events over the text above 
</div> 
</body>
</html>
